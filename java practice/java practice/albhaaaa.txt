class Main {
    public static void main(String[] args) {
        String name = "aabcc";
        String ans ="";
        int i=0;
        int count = 1;
        for(int j=1;j<name.length();j++){
            if(name.charAt(j)!=name.charAt(i)){
                ans+=name.charAt(i);
                ans+=count;   //ithula yen separete add pantromnaa char kooda int add pannum pothu aschii value add aaguthu so first charater ah string la add panitu next int add pantrom
                i++;
                count = 1;
            }
            else{
                count++;
                i++;
            }
        }
        ans+=name.charAt(i);
        ans+=count;
    }
}
