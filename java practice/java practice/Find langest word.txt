FIND LANGEST WORD

class Main {
    public static void main(String[] args) {
       String word = "I am learning Java programming";
       String[] arr = word.trim().split("\\s+");
       String ans = "";
       for(int i=0;i<arr.length;i++){
           if(arr[i].length()>ans.length()){
               ans=arr[i];
           }
       }
       System.out.println(ans);
    }
}
