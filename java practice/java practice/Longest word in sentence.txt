class Main {
    public static void main(String[] args) {
        String word = "This is a programming interview preparation question";
        String ans = "";
        String temp = "";

        for(int i = 0; i < word.length(); i++) {
            if(word.charAt(i) == ' ') {
                if(temp.length() > ans.length()) {
                    ans = temp;
                }
                temp = "";  // reset temp for next word
            } else {
                temp += word.charAt(i);
            }
        }

        // After loop ends, check last word
        if(temp.length() > ans.length()) {
            ans = temp;
        }

        System.out.println("Longest word : " + ans);
        System.out.println("Length : " + ans.length());
    }
}
