public class Makegood {
        public String makeGood(String s) {
            char[] stack = new char[s.length()];
            int top = 0;
            for(int i=0;i<s.length();i++){
                char ch = s.charAt(i);
                if(top>0 && stack[top-1]-ch==32){
                    top--;
                }
                else{
                    stack[top] = ch;
                    top++;
                }
            }
            char[] a = Arrays.copyOf(stack,top);
            return new String(a);
        }
    }
