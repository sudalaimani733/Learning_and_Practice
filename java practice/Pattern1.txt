public class Main
{
	public static void main(String[] args) {
		for(int i=1;i<=2*5-1;i++){
		    int cols = i>5?2*5-i:i;
		    for(int j=1;j<=cols;j++){
		        System.out.print("* ");
		    }
		    System.out.println();
		}
	}
}
