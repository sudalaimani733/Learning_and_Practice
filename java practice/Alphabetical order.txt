REARRANGE ALPHABETICAL ORDER 

class Main {
    public static void main(String[] args) {
        String word = "mAni";
        String ans = "";

        // Loop from 'a' to 'z'
        for (char ch = 'a'; ch <= 'z'; ch++) {
            for (int i = 0; i < word.length(); i++) {
                // Compare in lowercase
                if (Character.toLowerCase(word.charAt(i)) == ch) {
                    ans += word.charAt(i);  // Keep original case
                }
            }
        }

        System.out.println(ans);  // Output: Aimn
    }
}
